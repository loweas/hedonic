Code 
hawaii=read_csv("/Users/ashleylowe/Hawaii - Recreation/Trail/hawaii.csv")


hawaii_quantiles <- hawaii %>%
  group_by(state_name) %>%
  summarize(
    Mean = round(mean(popularity, na.rm = TRUE), 2),
    Q50 = round(quantile(popularity, 0.50, na.rm = TRUE), 2),
    Q75 = round(quantile(popularity, 0.75, na.rm = TRUE), 2),
    Q90 = round(quantile(popularity, 0.90, na.rm = TRUE), 2),
    AverageUsage = round(mean(visitor_usage, na.rm = TRUE), 2)
  )

# Interactive HTML table
datatable(
  hawaii_quantiles,
  caption = "Popularity Mean, Median and Visitor Usage Average for Hawaii by Island",
  options = list(pageLength = 10, autoWidth = TRUE)
)

Code 
poptrails=hawaii[c(4,5,7,8,9,10,11,12,34,35)]
poptrails <- st_as_sf(poptrails, coords = c("_geoloc.lng", "_geoloc.lat"), crs = 4326)
tmap_mode("view")
center_lon <- -157.45
center_lat <- 20.75
zoom_level <- 7

tm_shape(poptrails) +
    tm_dots(
        col = "popularity",       # color based on popularity
        palette = "viridis",
        title = "Popularity",
        size = 0.01,              # adjust point size
        id = "name",              # hover info
        popup.vars = c(
            "Popularity" = "popularity"
        )
    ) +
    tm_view(set.view = c(center_lon, center_lat, zoom_level))
Code 
# 1. find nearest feature index
nearest_idx <- st_nearest_feature(PB_HawaiiInfo, poptrails)

# 2. get nearest trails (same row order as PB_HawaiiInfo)
nearest_trails <- poptrails[nearest_idx, ]

# 3. calculate distance (element-wise)
nearest_dist_m <- st_distance(PB_HawaiiInfo, nearest_trails, by_element = TRUE)

# 4. merge trail attributes + distance into PB_HawaiiInfo
PB_HawaiiInfo <- PB_HawaiiInfo %>%
  mutate(
    nearest_trail_id    = nearest_idx,                         # row index in poptrails
    nearest_trail_meters = as.numeric(nearest_dist_m)
  ) %>%
  bind_cols(st_drop_geometry(nearest_trails))  # adds trail attributes
